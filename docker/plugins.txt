# =============================================================================
# Discourse plugins reference
# =============================================================================
#
# As of Discourse v2026.x, ALL plugins we need ship with core in the plugins/
# directory. No third-party cloning is required. This file is documentation
# only — it is not consumed by the Dockerfile.
#
# Plugins are present in the image but disabled by default. Enable them via
# the admin UI at /admin/plugins or via site settings API.
# =============================================================================

# --- Plugins we use (all bundled with Discourse core) ---
#
# plugins/discourse-openid-connect   # Keycloak SSO via OIDC (discovery, PKCE, RP-initiated logout)
# plugins/discourse-topic-voting     # Category-level feature voting with vote pools
# plugins/discourse-solved           # Mark replies as accepted solutions
# plugins/discourse-assign           # Assign topics/posts to staff members
# plugins/discourse-ai               # AI features (summarize, classify, embeddings)
# plugins/discourse-reactions        # Post reactions beyond simple likes
# plugins/automation                 # Automate actions via triggers and scripts
# plugins/chat                       # Real-time chat channels, DMs, threads
# plugins/poll                       # Create polls in posts
# plugins/checklist                  # Checkbox lists in posts
# plugins/discourse-details          # Collapsible detail/summary blocks
# plugins/discourse-narrative-bot    # New user tutorial bot
# plugins/discourse-presence         # "user is typing" indicators
# plugins/styleguide                 # Admin style guide for theming

# --- If a future plugin is NOT in core, add it to the Dockerfile: ---
# RUN cd plugins && \
#     sudo -u discourse git clone --depth 1 https://github.com/org/plugin-name.git
